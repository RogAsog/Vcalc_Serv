27// Вспомогательная функция для преобразования vector<string> в char* argv[]
28vector<char*> create_argv(const vector<string>& args) {
29 vector<char*> argv;
30 argv.push_back(const_cast<char*>("server")); // argv[0]
31
32 for (const auto& arg : args) {
33 char* arg_copy = new char[arg.size() + 1];
34 strcpy(arg_copy, arg.c_str());
35 argv.push_back(arg_copy);
36 }
37
38 argv.push_back(nullptr); // NULL terminator
39 return argv;
40}
