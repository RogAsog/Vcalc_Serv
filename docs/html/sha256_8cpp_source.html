101void sha256(const uint8_t *data, size_t len, uint8_t out[32]) {
102 // Инициализация начальных хэш-значений
103 uint32_t h[8] = {
104 0x6a09e667UL,0xbb67ae85UL,0x3c6ef372UL,0xa54ff53aUL,
105 0x510e527fUL,0x9b05688cUL,0x1f83d9abUL,0x5be0cd19UL
106 };
107
108 // Подготовка сообщения (дополнение до кратного 512 битам)
109 size_t num_blocks = ((len + 9) + 63) / 64;
110 uint8_t *msg = (uint8_t*)calloc(num_blocks*64, 1);
111 memcpy(msg, data, len);
112
113 // Добавление бита '1'
114 msg[len] = 0x80;
115
116 // Добавление длины сообщения в битах (64 бита, старший байт вперед)
117 uint64_t bit_len = (uint64_t)len * 8;
118 uint8_t *p = msg + num_blocks*64 - 8;
119 for (int i = 7; i >= 0; i--) {
120 p[i] = bit_len & 0xFF;
121 bit_len >>= 8;
122 }
123
124 // Обработка каждого 512-битного блока
125 for (size_t b = 0; b < num_blocks; b++) {
126 uint32_t w[64];
127 const uint8_t *chunk = msg + b*64;
128
129 // Подготовка расписания сообщений
130 for (int t = 0; t < 16; t++) {
131 w[t] = (chunk[t*4] << 24) | (chunk[t*4+1] << 16) |
132 (chunk[t*4+2] << 8) | (chunk[t*4+3]);
133 }
134
135 for (int t = 16; t < 64; t++) {
136 w[t] = ssig1(w[t-2]) + w[t-7] + ssig0(w[t-15]) + w[t-16];
137 }
138
139 // Инициализация рабочих переменных
140 uint32_t a = h[0];
141 uint32_t b0 = h[1];
142 uint32_t c = h[2];
143 uint32_t d = h[3];
144 uint32_t e = h[4];
145 uint32_t f = h[5];
146 uint32_t g = h[6];
147 uint32_t hh = h[7];
148
149 // Основной цикл сжатия
150 for (int t = 0; t < 64; t++) {
151 uint32_t T1 = hh + bsig1(e) + ch(e, f, g) + k[t] + w[t];
152 uint32_t T2 = bsig0(a) + maj(a, b0, c);
153 hh = g;
154 g = f;
155 f = e;
156 e = d + T1;
157 d = c;
158 c = b0;
159 b0 = a;
160 a = T1 + T2;
161 }
162
163 // Обновление хэш-значений
164 h[0] += a;
165 h[1] += b0;
166 h[2] += c;
167 h[3] += d;
168 h[4] += e;
169 h[5] += f;
170 h[6] += g;
171 h[7] += hh;
172 }
173
174 // Освобождение памяти
175 free(msg);
176
177 // Преобразование хэш-значений в байтовый массив (big-endian)
178 for (int i = 0; i < 8; i++) {
179 out[i*4] = (h[i] >> 24) & 0xFF;
180 out[i*4+1] = (h[i] >> 16) & 0xFF;
181 out[i*4+2] = (h[i] >> 8) & 0xFF;
182 out[i*4+3] = (h[i]) & 0xFF;
183 }
184}
